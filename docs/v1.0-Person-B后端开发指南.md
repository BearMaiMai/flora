### 0.2 现有 Service 层调用方式

项目已有 `utils/cloud.js` 封装，各 service 文件基于它调用云函数：

```javascript
// miniprogram/utils/cloud.js 已有封装
const callFunction = async (name, data = {}, options = {}) => {
  // 自动处理 loading、错误日志
  const res = await wx.cloud.callFunction({ name, data })
  return res.result  // 直接返回 result，不做 code 判断
}

// 各 service 使用方式（以 flower.js 为例）
const { callFunction } = require('../utils/cloud')
const flowerService = {
  getList(params = {}) {
    return callFunction('flower', { action: 'list', ...params })
  },
  // ...
}
```

> ⚠️ **注意**：`cloud.js` 返回的是 `res.result`（即云函数 return 的整个对象，包含 `code` 和 `data`）。  
> 页面中使用时要取 `result.data`，或在 service 层统一处理。

### 0.3 云函数结构约定

所有云函数统一采用「内部路由分发」模式：

```javascript
// 前端调用格式
wx.cloud.callFunction({
  name: '云函数名',         // 如 'flower'
  data: {
    action: 'list',        // 操作名
    // ...其他参数
  }
})

// 云函数返回格式（统一）
{ code: 0, data: { /* 数据 */ } }           // 成功
{ code: -1, message: '错误信息' }           // 失败
```

```javascript
/**
 * services/common.js - 公共数据接口

const { callFunction } = require('../utils/cloud')

const commonService = {
  /** 获取首页聚合数据（推荐花卉 + 每日贴士 + 今日提醒） */
```

### 各 Service 文件调用方式速查

页面中引入 service 的标准写法：

```javascript
// 在页面 .js 文件中（以首页为例）
const commonService = require('../../services/common')
const flowerService = require('../../services/flower')

Page({
  async onLoad() {
    try {
      const result = await commonService.getHomeData()
      // result 格式：{ code: 0, data: { todayRecommend: [...], dailyTip: {...} } }
      if (result.code === 0) {
        this.setData({
          recommendList: result.data.todayRecommend,
          dailyTip: result.data.dailyTip,
        })
      }
    } catch (err) {
      console.error('首页数据加载失败', err)
      wx.showToast({ title: '加载失败，请重试', icon: 'error' })
    }
  }
})
```

```javascript
const flowerService = require('../../services/flower')

Page({
  data: {
    loading: true,
    flowerList: [],
    searchKeyword: '',
    currentCategory: 'all',
  },

  async onLoad() {
    this.loadFlowerList()
  },

  async loadFlowerList() {
    this.setData({ loading: true })
    const params = {}
    if (this.data.currentCategory !== 'all') {
      params.category = this.data.currentCategory
    }
    try {
      const result = await flowerService.getList(params)
      if (result.code === 0) {
        this.setData({ flowerList: result.data.list })
      }
    } finally {
      this.setData({ loading: false })
    }
  },

  // 搜索
  async onSearch(e) {
    const keyword = e.detail.value.trim()
    if (!keyword) return this.loadFlowerList()
    const result = await flowerService.search(keyword)
    if (result.code === 0) {
      this.setData({ flowerList: result.data.list })
    }
  },

  // 分类切换
  onCategoryChange(e) {
    this.setData({ currentCategory: e.detail.value }, () => {
      this.loadFlowerList()
    })
  },
})
```

### 百科列表分页实现

```javascript
Page({
  data: {
    flowerList: [],
    page: 1,
    pageSize: 20,
    total: 0,
    hasMore: true,
    loadingMore: false,
  },

  async onLoad() {
    this.loadFlowerList(true)  // true = 初始加载，重置列表
  },

  // 下拉刷新
  async onPullDownRefresh() {
    await this.loadFlowerList(true)
    wx.stopPullDownRefresh()
  },

  // 上拉加载更多
  async onReachBottom() {
    if (!this.data.hasMore || this.data.loadingMore) return
    await this.loadFlowerList(false)  // false = 追加
  },

  async loadFlowerList(reset = false) {
    if (reset) {
      this.setData({ page: 1, flowerList: [], hasMore: true })
    }

    const { page, pageSize } = this.data
    this.setData({ loadingMore: true })

    try {
      const result = await flowerService.getList({ page, pageSize })
      if (result.code === 0) {
        const { list, total } = result.data
        const newList = reset ? list : [...this.data.flowerList, ...list]
        this.setData({
          flowerList: newList,
          total,
          page: page + 1,
          hasMore: newList.length < total,
        })
      }
    } finally {
      this.setData({ loadingMore: false })
    }
  },
})
```

```javascript
async function getHomeData(params, db) {
  const { _openid } = params

  // 并发查询，提高性能
  const [recommendRes, tipRes, seasonRes] = await Promise.all([
    // 今日推荐：取前3条排序靠前的已发布花卉
    db.collection('flowers')
      .where({ isPublished: true })
      .orderBy('sortOrder', 'asc')
      .limit(3)
      .field({ _id: true, name: true, coverImage: true, difficulty: true, category: true })
      .get(),

    // 每日贴士：按日期取一条（用当日第几天 mod 贴士总数）
    db.collection('daily_tips')
      .where({ isPublished: true })
      .orderBy('sortOrder', 'asc')
      .limit(10)
      .get(),

    // 当季花卉：取当前月份对应季节的花卉
    db.collection('flowers')
      .where({ isPublished: true })
      .orderBy('favoriteCount', 'desc')
      .limit(6)
      .field({ _id: true, name: true, coverImage: true, difficulty: true })
      .get(),
  ])

  // 选取今日贴士（按日期轮播）
  const tips = tipRes.data
  const dayOfYear = Math.floor((Date.now() - new Date(new Date().getFullYear(), 0, 0)) / 86400000)
  const dailyTip = tips[dayOfYear % tips.length] || tips[0]

  // 查询用户今日提醒（已登录时）
  let myReminders = []
  if (_openid) {
    const today = new Date()
    today.setHours(23, 59, 59, 999)
    const reminderRes = await db.collection('reminders')
      .where({
        _openid,
        isCompleted: false,
        nextRemindAt: db.command.lte(today),
      })
      .limit(5)
      .get()
    myReminders = reminderRes.data
  }

  return {
    todayRecommend: recommendRes.data,
    dailyTip,
    seasonFlowers: seasonRes.data,
    myReminders,
  }
}

module.exports = getHomeData
```
